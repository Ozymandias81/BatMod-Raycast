ACTOR BatFist : Weapon
{
	Weapon.SelectionOrder 3700
	Weapon.Kickback 100
	Obituary "$OB_MPFIST"
	Tag "$TAG_FIST"
	AttackSound "*fist"
	//Scale 3
	+WEAPON.MELEEWEAPON
	+NoAlert
	States
	{
	Spawn:
		WPIC B -1
		Stop
	Ready:
		FIST A 1 A_WeaponReady
		Loop
	Deselect:
		FIST A 1 A_Lower
		Loop
	Select:
		FIST A 1 A_Raise
		Loop
	AltFire:
		BKIK A 3 
		BKIK B 3
		BKIK C 3 A_CustomPunch(4 * random(1, 10), FALSE, 0)
		BKIK DE 2 A_ReFire
		Goto Ready
	Fire:
		FIST B 4 A_CustomPunch(4 * random(1, 10), FALSE, 0)
		FIST A 3
		FIST C 4 A_CustomPunch(4 * random(1, 10), FALSE, 0)
		FIST A 2 A_ReFire
		Goto Uppercut
	Uppercut:
		FIST E 4 A_CustomPunch(6 * random(1, 10), FALSE, 0)
		FIST F 3 A_CustomPunch(6 * random(1, 10), FALSE, 0)
		FIST E 2 
		Goto Ready
	}
}
